static T genderLogic<T>(String targetGender,
    {T? female, T? male, required T other, String? locale}) {
  ArgumentError.checkNotNull(other, 'other');
  switch (targetGender) {
    case 'female':
      return female == null ? other : female;
    case 'male':
      return male == null ? other : male;
    default:
      return other;
  }
}
