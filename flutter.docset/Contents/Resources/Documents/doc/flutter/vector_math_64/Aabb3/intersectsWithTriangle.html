  double a;

  // define a aabb
  copyCenterAndHalfExtents(_aabbCenter, _aabbHalfExtents);

  // Translate triangle as conceptually moving AABB to origin
  _v0
    ..setFrom(other.point0)
    ..sub(_aabbCenter);
  _v1
    ..setFrom(other.point1)
    ..sub(_aabbCenter);
  _v2
    ..setFrom(other.point2)
    ..sub(_aabbCenter);

  _f2
