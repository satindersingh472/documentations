@override
CupertinoIconThemeData copyWith({
  double? size,
  double? fill,
  double? weight,
  double? grade,
  double? opticalSize,
  Color? color,
  double? opacity,
  List<Shadow>? shadows,
}) {
  return CupertinoIconThemeData(
    size: size ?? this.size,
    fill: fill ?? this.fill,
    weight: weight ?? this.weight,

    color: color ?? this.color,
    opacity: opacity ?? this.opacity,
    shadows: shadows ?? this.shadows,
  );
}
